[cbenjamin@node1 ~]$ cat /etc/nginx/nginx.conf
worker_processes 1;
error_log nginx_error.log;
events {
    worker_connections 1024;
}

http {
     server {
        listen 80;

        server_name node1.tp1.b2;

        location / {
                return 301 /site1;
        }

        location /site1 {
                alias /srv/site1;
        }

        location /site2 {
                alias /srv/site2;
        }
}

server {
        listen 443 ssl;

        server_name node1.tp1.b2;
        ssl_certificate server.crt;
        ssl_certificate_key server.key;

        location / {
            return 301 /site1;
        }

        location /site1 {
            alias /srv/site1;
        }
        location /site2 {
            alias /srv/site2;
        }
    }
}
[cbenjamin@node1 ~]$
